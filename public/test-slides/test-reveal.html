// 인사말 생성 함수
const greeting = 'Hello, Reveal.js!';

function greet(name) {
  const message = `${greeting} 안녕하세요, ${name}님!`;
  console.log(message);
  return message;
}

const users = ['철수', '영희', '민준'];
users.forEach(user => greet(user));

이것은 발표자 노트입니다. 청중에게는 보이지 않고 발표자 화면에만 표시됩니다. 주요 테스트 항목 요약: - 슬라이드 1: 타이틀 및 JS 클릭 카운터 버튼 - 슬라이드 2: Fragment fade-in / highlight-red - 슬라이드 3a: 코드 블록 (highlight.js) - 슬라이드 3b: 데이터 테이블 - 슬라이드 4: 배경색, 체크박스 인터랙션 - 슬라이드 5: 최종 요약 및 발표자 노트 (현재 화면) SoEZ HTML 에디터에서 이 슬라이드 덱을 로드하면 슬라이드 덱으로 감지되어 슬라이드 네비게이션이 활성화됩니다.
